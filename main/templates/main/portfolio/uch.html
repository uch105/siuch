Data Visualisation: VSM

A VSM (Vibrating Sample Magnetometer) machine provides data with a .ASC extension file format, which is a bit confusing to handle. This simple piece of code converts the .ASC file to graphical view as png file. It becomes more handy when you are working with multiple samples and need to compare them. Check out my git repository.

                    
import matplotlib.pyplot as plt
import csv

with open('you_file_name.ASC', 'r') as asc_file, open('target_file_name.csv', 'w', newline='') as csv_file:
csv_writer = csv.writer(csv_file)

for line in asc_file:
    numbers = line.strip().split(',')
    csv_writer.writerow([float(numbers[0]), float(numbers[1])])

x_data = []
y_data = []

with open('target_file_name.csv', 'r') as csv_file:
    csv_reader = csv.reader(csv_file)
    for row in csv_reader:
        x_data.append(float(row[0]))
        y_data.append(float(row[1]))

plt.figure(figsize=(8, 8))
plt.plot(x_data, y_data, marker='.', linestyle='-', color='b')

plt.xlabel('Magnetic Field (Oe)')
plt.ylabel('Magnetization (emu/gm)')
plt.title('VSM Data Visualization')

plt.axhline(0, color='black',linewidth=0.5)
plt.axvline(0, color='black',linewidth=0.5)
plt.grid(True)

plt.savefig('vsm.jpg', format='jpg', dpi=300)
